Expo UI DisclosureGroup 与官方 SwiftUI [DisclosureGroup API](https://developer.apple.com/documentation/swiftui/disclosuregroup) 保持一致，并显示一个披露指示器，用于展开或隐藏内容。

## 用法

### 基本披露组

```tsx BasicDisclosureGroupExample.tsx
import { Host, DisclosureGroup, Text } from '@expo/ui/swift-ui';

export default function BasicDisclosureGroupExample() {
  return (
    <Host matchContents>
      <DisclosureGroup label="更多信息">
        <Text>当披露组展开时，将显示此内容。</Text>
      </DisclosureGroup>
    </Host>
  );
}
```

### 初始展开

将 `isExpanded` 初始设置为 `true`，即可默认显示内容。

```tsx InitiallyExpandedExample.tsx
import { useState } from 'react';
import { Host, DisclosureGroup, Text } from '@expo/ui/swift-ui';

export default function InitiallyExpandedExample() {
  const [isExpanded, setIsExpanded] = useState(true);

  return (
    <Host matchContents>
      <DisclosureGroup label="详情" isExpanded={isExpanded} onIsExpandedChange={setIsExpanded}>
        <Text>此内容默认可见。</Text>
      </DisclosureGroup>
    </Host>
  );
}
```
