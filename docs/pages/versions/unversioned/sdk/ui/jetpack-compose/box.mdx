Expo UI Box 与官方 Jetpack Compose [Box](https://developer.android.com/reference/kotlin/androidx/compose/foundation/layout/package-summary#Box) API 一致，并支持通过可配置的内容对齐方式将子元素叠放在彼此之上。

## 用法

`Box` 会将子元素叠放在彼此之上。使用 `contentAlignment` 来在 Box 中定位它们。

```tsx BoxExample.tsx
import { Host, Box, Text } from '@expo/ui/jetpack-compose';
import { size, background } from '@expo/ui/jetpack-compose/modifiers';

export default function BoxExample() {
  return (
    <Host matchContents>
      <Box contentAlignment="center" modifiers={[size(200, 200), background('#E0E0E0')]}>
        <Text>在 Box 中居中</Text>
      </Box>
    </Host>
  );
}
```
